########################################################################
# Check if there is C++ code at all
########################################################################
if(NOT si446x_sources)
    MESSAGE(STATUS "No C++ sources... skipping python bindings")
    return()
endif(NOT si446x_sources)

########################################################################
# Check for pygccxml
########################################################################
GR_PYTHON_CHECK_MODULE_RAW(
    "pygccxml"
    "import pygccxml"
    PYGCCXML_FOUND
    )

include(GrPybind)

########################################################################
# Python Bindings
########################################################################

list(APPEND si446x_python_files
    python_bindings.cc)

GR_PYBIND_MAKE_OOT(si446x
   ../../..
   gr::si446x
   "${si446x_python_files}")

# copy bindings extension for use in QA test module
add_custom_command(TARGET si446x_python POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy $<TARGET_FILE:si446x_python>
    ${PROJECT_BINARY_DIR}/test_modules/gnuradio/si446x/
)

install(TARGETS si446x_python DESTINATION ${GR_PYTHON_DIR}/gnuradio/si446x COMPONENT pythonapi)
